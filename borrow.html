# หน้ายืม @app.route('/borrow/', methods=['GET', 'POST']) def borrow(id): conn = connect_db() cursor = conn.cursor() if request.method == 'POST': cursor.execute('UPDATE equipment SET available = 0 WHERE id = ?', (id,)) conn.commit() conn.close() return redirect(url_for('index')) else: cursor.execute('SELECT * FROM equipment WHERE id = ?', (id,)) data = cursor.fetchone() conn.close() return render_template('borrow.html', equipment=data)
